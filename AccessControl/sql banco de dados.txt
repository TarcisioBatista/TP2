CREATE TABLE user(
idUser INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
login VARCHAR(20) NOT NULL,
senha VARCHAR(8) NOT NULL
);


CREATE TABLE descAcao(
idDescAcao INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
descricao VARCHAR(100) NOT NULL
);


CREATE TABLE acao(
idAcao INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
idDescAcao INTEGER REFERENCES descAcao(idDescAcao) on delete cascade,
idUser INTEGER REFERENCES user(idUser) on delete cascade,
data TEXT,
hora TEXT
);



CREATE TABLE log(
idLog INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
idAcao INTEGER REFERENCES acao(idAcao) on delete cascade
);
